  (use-package ido
    :init
      (progn
        (defun ido-M-x ()
          (interactive)
            (call-interactively

                  "M-x "
                    (all-completions "" obarray 'commandp)))))

    (ido-mode 1)
    (setq ido-enable-flex-matching t)
